SCROLL TO TOP

Click to scroll down
Click to scroll up

How to use it:

Add Html


    <!DOCTYPE html>

    <html>

    <body>

    

     <div class="wrapper scroll-to-top">
        <a href="#bottom" id="top">Click to scroll down</a>
        <div id="bottom" class="star"></div>
        <a href="#top1" >Click to scroll up</a>
        <div id="top1" class="star1"></div>
     </div>
    

    </body>

    </html>

Add Css

  
    
     .wrapper{
        width: 100%;
        height: 400px;
        display: inline-block;
        background: #f5f2f0;
        position: relative;
        overflow-y: scroll;
      }

      .scroll-to-top a {
        background: #ddd;
        color: #444;
        font-family: sans-serif;
        text-align: center;
        text-decoration: none;
        padding: 20px;
        display: block;
      }

      .scroll-to-top #star {
        text-align: center;
        font-size: 100px;
        line-height: 300px;
        color: #ddd;
      }
      .scroll-to-top .star{
        height:400px;
        width:100%;
        background:#444;
      }
      .scroll-to-top .star1{
        height:400px;
        width:100%;
        background:#000;
      }

Add Jquery & Script

  
    <script src="js/jquery.min.js"></script>
    <script>
      $('a[href^="#"]').on('click', function(event) {

        var target = $(this.getAttribute('href'));

        if( target.length ) {
            event.preventDefault();
            $('div').stop().animate({
                scrollTop: target.offset().top
            }, 1000);
        }

    });

    </script>
