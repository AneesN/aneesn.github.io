BUTTON CLICK SLIDE DIV

How to use it:

Add Html


    <!DOCTYPE html>

    <html>

    <body>

    

     <div class="wrapper">
       <div class="header_block"> Header Fixed</div>
       <div class="bo_dy"></div>
     </div>
    

    </body>

    </html>

Add Css

  
    
     .wrapper{
        width: 100%;
        height: 400px;
        display: inline-block;
        background: #f5f2f0;
        position: relative;
        overflow-y: scroll;
      }

      .header_block{
        width:100%;
        height:70px;
        background:#000;
        position: sticky;
        top:0px;
        left:0px;
        -webkit-transition: 1s;
          -moz-transition: 1s;
          -ms-transition: 1s;
          -o-transition: 1s;
          transition: 1s;
        color:#fff;
        font-size:32px;
        text-align:center;
        line-height:70px;
        
      }
      .fixedHeader{
       
         top:-150px;
        -webkit-transition: 1s;
          -moz-transition: 1s;
          -ms-transition: 1s;
          -o-transition: 1s;
          transition: 1s;
      }
      .scrollTop{
         top:0px;
        -webkit-transition: 1s;
          -moz-transition: 1s;
          -ms-transition: 1s;
          -o-transition: 1s;
          transition: 1s;
          
      }


      .bo_dy{
        width:100%;
        height:1000px;
        background: #f5f2f0;
      }

Add Jquery & Script

  
    <script src="js/jquery.min.js"></script>
    <script>
      function fxHeader(){

      var wnWdth = jQuery('.wrapper').width();
        if (wnWdth > 500){
          jQuery('.wrapper').scroll(function() {
                  var wnht = jQuery('.wrapper').height();
                  var scroll = jQuery('.wrapper').scrollTop();

                  if (scroll >= 50) {
                      jQuery(".header_block").addClass("fixedHeader");
                  } else {
                      jQuery(".header_block").removeClass("fixedHeader");
                  }

                  if (scroll >= wnht) {
                      jQuery(".header_block").addClass("scrollTop");
                  } else {
                      jQuery(".header_block").removeClass("scrollTop");
                  }

              });

        }
      }


      fxHeader();

    </script>
