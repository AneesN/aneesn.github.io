Add Jquery & Script

  

      <script src="js/jquery.min.js"></script>

      <script>

        jQuery(document).ready(function() {

            if (jQuery('.md-input>input, .md-input>textarea').length > 0) {
                jQuery('.md-input>input, .md-input>textarea').each(function() {
                    if (jQuery(this).val()) {
                        jQuery(this).parent().addClass('active');
                    }
                });

                jQuery('.md-input>input, .md-input>textarea').focusin(function() {
                    jQuery(this).parent().addClass('active');

                });
                jQuery('.md-input>input, .md-input>textarea').focusout(function() {

                     if($(this).val()==""){ 
                         jQuery(this).parent().removeClass('active');
                     } 

                });
            }

        });


      </script>
